## 一、技术栈推荐


## 五、关键功能实现方案

### 5.1 摇卦算法（硬币法）

```typescript
// lib/iching.ts

interface Line {
  value: number;     // 6, 7, 8, 9
  isYin: boolean;    // true = 阴爻, false = 阳爻
  isChanging: boolean; // 老阴(6)或老阳(9)为变爻
}

interface HexagramResult {
  primary: number;   // 本卦编号 (1-64)
  changing: number[]; // 变爻位置 [1-6]
  transformed?: number; // 变卦编号
  lines: Line[];
}

// 三枚硬币，正面=3，反面=2
// 总和: 6(老阴), 7(少阳), 8(少阴), 9(老阳)
function castLine(): Line {
  const coins = [
    Math.random() > 0.5 ? 3 : 2,
    Math.random() > 0.5 ? 3 : 2,
    Math.random() > 0.5 ? 3 : 2,
  ];
  const sum = coins.reduce((a, b) => a + b, 0);
  
  return {
    value: sum,
    isYin: sum === 6 || sum === 8,
    isChanging: sum === 6 || sum === 9,
  };
}

function castHexagram(): HexagramResult {
  const lines: Line[] = [];
  for (let i = 0; i < 6; i++) {
    lines.push(castLine());
  }
  
  const primary = linesToNumber(lines);
  const changing = lines
    .map((l, i) => l.isChanging ? i : -1)
    .filter(i => i !== -1);
  
  const transformed = changing.length > 0
    ? linesToNumber(lines.map(l => ({
        ...l,
        isYin: l.isChanging ? !l.isYin : l.isYin,
      })))
    : undefined;
  
  return { primary, changing, transformed, lines };
}

// 卦象查找表（64卦的二进制映射）
// 阳爻=1, 阴爻=0, 从下往上读
const HEXAGRAM_MAP: Record<string, number> = {
  '111111': 1,  // ䷀ 乾
  '000000': 2,  // ䷁ 坤
  '100010': 3,  // ䷂ 屯
  // ... 其余61卦
};

function linesToNumber(lines: Line[]): number {
  const binary = lines.map(l => l.isYin ? '0' : '1').join('');
  return HEXAGRAM_MAP[binary] || 1;
}
```

## 六、64卦数据结构（核心内容）

```typescript
interface Hexagram {
  id: number;                    // 1-64
  number: number;                // 序号
  name: string;                  // 英文名 "The Creative"
  chinese: string;               // 中文名 "乾"
  pinyin: string;                // 拼音 "Qián"
  unicode: string;               // Unicode 卦象符号 "䷀"
  trigrams: {
    upper: string;               // 上卦 "Heaven"
    lower: string;               // 下卦 "Heaven"
    chinese_upper: string;       // "天"
    chinese_lower: string;       // "天"
  };
  keywords: string[];            // ["Strength", "Initiative", "Power"]
  judgment: string;              // 卦辞（英文）
  image: string;                 // 象辞（英文）
  interpretation: {
    general: string;             // 通用解读
    love: string;                // 感情解读
    career: string;              // 事业解读
    growth: string;              // 成长解读
  };
  lines: {
    position: number;            // 1-6
    text: string;                // 爻辞
    interpretation: string;      // 爻辞解读
  }[];
  related: {
    opposite: number;            // 错卦
    reverse: number;             // 综卦
    nuclear: number;             // 互卦
  };
}
```
